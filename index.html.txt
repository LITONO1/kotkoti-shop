/* FORM TOGGLE */
function showOrderForm(btn){
let form = btn.nextElementSibling;
form.style.display = (form.style.display === "block") ? "none" : "block";
}

/* ORDER SUBMIT */
function submitOrder(btn){

let form = btn.parentElement;

let name = form.querySelector(".customer-name").value;
let phone = form.querySelector(".customer-phone").value;
let address = form.querySelector(".customer-address").value;

let productName = form.parentElement.querySelector(".card-title").innerText;

if(!name || !phone || !address){
alert("সব তথ্য পূরণ করুন");
return;
}

let orderId = Date.now();

db.ref("orders/" + orderId).set({
product: productName,
name,
phone,
address,
time: new Date().toLocaleString()
});

alert("অর্ডার সফল হয়েছে");

form.style.display = "none";

form.querySelector(".customer-name").value = "";
form.querySelector(".customer-phone").value = "";
form.querySelector(".customer-address").value = "";
}
